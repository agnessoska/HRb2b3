import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface OnboardingStore {
  completedTours: string[];
  skippedTours: string[];
  
  markCompleted: (tourId: string) => void;
  markSkipped: (tourId: string) => void;
  reset: () => void;
}

export const useOnboardingStore = create<OnboardingStore>()(
  persist(
    (set) => ({
      completedTours: [],
      skippedTours: [],
      
      markCompleted: (tourId) =>
        set((state) => ({
          completedTours: [...new Set([...state.completedTours, tourId])],
        })),
        
      markSkipped: (tourId) =>
        set((state) => ({
          skippedTours: [...new Set([...state.skippedTours, tourId])],
        })),
        
      reset: () => set({ completedTours: [], skippedTours: [] }),
    }),
    { name: 'onboarding-progress' }
  )
);

### 8.3 –•—É–∫ useOnboarding

```typescript
// useOnboarding.ts
import { driver, type Driver } from 'driver.js';
import 'driver.js/dist/driver.css';
import { useCallback, useRef } from 'react';
import { useTranslation } from 'react-i18next';
import { useOnboardingStore } from './onboarding.store';
import { getTourConfig, getTourIdForRoute } from '../config';

export const useOnboarding = () => {
  const { t, i18n } = useTranslation('onboarding');
  const driverRef = useRef<Driver | null>(null);
  const { completedTours, markCompleted, markSkipped, reset } = useOnboardingStore();

  const startTour = useCallback((tourId: string) => {
    const tourConfig = getTourConfig(tourId, t, i18n.language);
    if (!tourConfig) return;

    // –°–æ–∑–¥–∞—ë–º —ç–∫–∑–µ–º–ø–ª—è—Ä driver
    driverRef.current = driver({
      showProgress: true,
      showButtons: ['next', 'previous', 'close'],
      nextBtnText: t('common.next'),
      prevBtnText: t('common.back'),
      doneBtnText: t('common.finish'),
      
      // –ö–∞—Å—Ç–æ–º–Ω—ã–π CSS –∫–ª–∞—Å—Å –¥–ª—è —Å—Ç–∏–ª–∏–∑–∞—Ü–∏–∏
      popoverClass: 'onboarding-popover',
      
      // –ü–ª–∞–≤–Ω—ã–π —Å–∫—Ä–æ–ª–ª
      smoothScroll: true,
      scrollIntoViewOptions: { behavior: 'smooth', block: 'center' },
      
      // Callbacks
      onDestroyStarted: () => {
        if (driverRef.current?.isLastStep()) {
          markCompleted(tourId);
        } else {
          markSkipped(tourId);
        }
        driverRef.current?.destroy();
      },
      
      steps: tourConfig.steps,
    });

    driverRef.current.drive();
  }, [t, i18n.language, markCompleted, markSkipped]);

  const stopTour = useCallback(() => {
    driverRef.current?.destroy();
    driverRef.current = null;
  }, []);

  return {
    startTour,
    stopTour,
    completedTours,
    isTourCompleted: (id: string) => completedTours.includes(id),
    resetProgress: reset,
    getTourIdForRoute,
  };
};
```
